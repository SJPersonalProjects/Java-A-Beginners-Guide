#if
if(condition) statement;
else statement;

#switch
switch(expression) {
	case constant:
		statement sequence
		break;
	//...
}

#for
for(initialization; condition; iteration) statement;

#while
while(condition) statement

#do-while
do{
	statement;
}while(condition);

#break
break; or break label;

#continue
continue; or continue label;
